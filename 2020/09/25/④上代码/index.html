发表于 2020-09-25 分类于 程序人生 ， 学习教育 阅读次数： 本文字数： 20k 阅读时长 ≈ 18 分钟

核心代码3大块：

  1. 微信中获取随申码
  2. 钉钉中登记并上传截图
  3. 钉钉换成哓黑板
import win32gui
import win32con

"""
本程序主要实现3个功能：
①登陆微信，找到随申办小程序，然后获得包括本人和亲属的随申码，并保存成jpg文件
②登陆钉钉，找到相应的组织机构，根据要求提交每日健康申报
③登陆晓黑板，根据班级要求，提交每日健康申报

本程序主要通过pywin32和pyautogui对应用的窗口进行定位，使用了图片比对和像素定位
两种方式，目前版本[v1.0]不支持文字识别。

主要存在的限制：
①必须安装微信、钉钉和晓黑板的PC客户端版本
②在微信中，必须先安装'随申办'小程序, 并注册用户；然后将亲属随申码都申请好
③在配置参数中：
  - familyList这个参数的名单：亲属顺序根据微信’随申码‘的顺序，本人(微信登录用户)在最后,根据钉钉的设置，最多5个人
④由于暂不支持文字识别，所以在定位控件的时候都是采用pyautogui的图片比较，在图片比较失效的情况下，采用默认定位位置，
  因此，如果需要更换一些参数，比如省-市-区，比如钉钉的组织的话，需要的操作如下：
      A) 钉钉的组织修改：
         方法A： 先登录钉钉，并将默认组织修改为小孩所在的班级
         方法B： 登录钉钉，将小孩所在的班级的图片截图下来，并命名为dd_org.png和dd_orgselected.png放在img文件夹下
      B) 钉钉日报提交的省-市-区的修改
         方法A： 登录钉钉，手工提交一次，以后钉钉会默认记住
         方法B： 登录钉钉，将自己所在的省市区截图，然后命名为
                 Sheng.png和Sheng_selected.png
                 Shi.png和Shi_selected.png
                 Qu.png和Qu_selected.png

已知BUG：
①在钉钉和晓黑板打开文件的对话框的时候，需要事先手工打开一次，把文件夹定位到本目录，确保寻找随申码文件不会失效
  *后续版本会考虑去除这个BUG
②微信如果打开的时候为扫码登录的话，应用可能会报错，但是只要扫码登录了，应用将继续正常运行
"""

# 日志
def getLogger(basicLogFileName, logFileName, logLevel=logging.DEBUG, logMode=1):

    # 日志格式
    logFormat = logging.Formatter('%(asctime)s - %(pathname)s[line:%(lineno)d] - %(levelname)s: %(message)s')
    # 定义日志
    logging.basicConfig(filename=basicLogFileName,
                        format='%(asctime)s - %(pathname)s[line:%(lineno)d] - %(levelname)s: %(message)s',
                        level=logLevel)
    logger = logging.getLogger(__name__)
    logger.setLevel(level=logLevel)

    if logMode == 0:
        ch = logging.StreamHandler()
        ch.setLevel(logLevel)
        ch.setFormatter(logFormat)
        logger.addHandler(ch)
    elif logMode == 1:
        fh = logging.FileHandler(logFileName)
        fh.setLevel(logLevel)
        fh.setFormatter(logFormat)
        logger.addHandler(fh)
    else:
        ch = logging.StreamHandler()
        ch.setLevel(logLevel)
        ch.setFormatter(logFormat)
        logger.addHandler(ch)    
        fh = logging.FileHandler(logFileName)
        fh.setLevel(logLevel)
        fh.setFormatter(logFormat)
        logger.addHandler(fh)

    return logger

def getPassword(key):
    db = shelve.open(cd.passwordFile)
    retValue = db[key]
    db.close()
    return retValue

# 把指定的窗口移动到新的位置，一般是0,0
def moveWindow(window, newLeft, newTop):
    win32gui.ShowWindow(window, win32con.SW_SHOWNORMAL)

    # 把窗口激活
    win32gui.SetForegroundWindow(window)

    left, top, right, bottom = win32gui.GetWindowRect(window)

    # 把窗口移动到（0，0）
    win32gui.MoveWindow(window, cd.left, cd.top, abs(right-left), abs(bottom-top), cd.repaint)
    time.sleep(cd.generalWaitTime)

# 根据不同的应用启动应用程序
def getAppStart(appName, mainWndClassName, mainWndName):
    '''
    通过参数appName，mainWndClassName和mainWndName，判断相应的应用是否已经启动:
    ①如果已经启动，直接返回该应用的主窗口的句柄；
    ②如果没有启动，根据传入的参数，启动该应用并返回该应用的主窗口的句柄；
    ③如果打开失败，抛出异常；

    参数:
        - appName: 绝对路径表示的应用名称
        - mainWndClassName: 应用主窗口的类名(可以通过spy++获取)
        - mainWndName: 应用主窗口名称
    返回值：
        - mainWnd: 主窗口的句柄
    异常：
        - OpenAppException
    '''

    logger.info("开始获取应用：{}".format(appName))

    # 获取应用主窗口
    # 晓黑板和钉钉只要没有主动退出，下一次打开会用默认账号登录，
    # 而微信每次打开都要单独登录
    mainWnd = win32gui.FindWindow(mainWndClassName, mainWndName)

    if mainWnd == 0:   # 主窗口没有被打开
        startApp(appName)    # 启动应用
        time.sleep(cd.generalWaitTime * cd.generalTimeAdjust)   # 等待主窗口显示

        if 'WeChat' in appName:
            logger.info('打开微信登录...')
            loginWindow = win32gui.FindWindow(cd.wxLoginWindowClass, cd.wxLoginWindowName)
            moveWindow(loginWindow, 0, 0)
            mainWnd = wxLogin()
        elif 'DingDing' in appName:
            logger.info('打开钉钉登录...')
            loginWindow = win32gui.FindWindow(cd.ddLoginWindowClass, cd.ddLoginWindowName)
            moveWindow(loginWindow, 0, 0)
            mainWnd = ddLogin()
        else:
            logger.info('打开晓黑板登录...')
            loginWindow = win32gui.FindWindow(cd.xhbLoginWindowClass, cd.xhbLoginWindowName)
            moveWindow(loginWindow, 0, 0)
            mainWnd = xhbLogin()

    logger.debug("返回主窗口句柄：{}".format(mainWnd))    
    return mainWnd

# 得到需要点击的位置
# Return: target_x,target_y
# x, y: 如果不能识别图片的话，使用
def getClickPosition(pngName, _x, _y):
    myLocation = pyautogui.locateOnScreen(pngName)
    if myLocation != None:
        logger.debug("找到了图片 {} 的位置，返回图片中央坐标".format(pngName))
        btnPosition = pyautogui.center(myLocation)
        target_x, target_y = btnPosition
    else:
        logger.debug("找不到图片 {} 的位置，使用指定的默认坐标".format(pngName))
        target_x = cd.left + _x
        target_y = cd.top + _y
    return target_x, target_y

# 打开微信的随申办，得到每个人的随申码，并保存在当前目录
def processWxSuishencode(wxWindow):

    logger.info("开始处理从微信获取随申码...")

    # 1.2如果窗口最小化了就把它恢复到正常
    win32gui.ShowWindow(wxWindow,win32con.SW_SHOWNORMAL)

    # 1.3把窗口激活
    win32gui.SetForegroundWindow(wxWindow)

    # 1.4 把微信窗口移动到（0，0），大小调整为800x600
    win32gui.MoveWindow(wxWindow, cd.left, cd.top, cd.width, cd.height, cd.repaint)
    time.sleep(cd.generalWaitTime)

    # 2. 点击小程序面板
    target_x, target_y = getClickPosition(cd.wx_xcxPic, cd.microPg_x, cd.microPg_y)
    pyautogui.click(target_x, target_y)
    time.sleep(cd.generalWaitTime)

    # 2. 点击随申办小程序
    target_x, target_y = getClickPosition(cd.wx_ssbPic, cd.ssb_x, cd.ssb_y)
    pyautogui.click(target_x, target_y)


    # 3.1 打开了随申办的新窗口
    logger.info("打开随申办小程序的窗口...")
    time.sleep(cd.generalWaitTime * cd.generalTimeAdjust)   # 打开小程序还是有点耗时的
    ssbWindow=win32gui.FindWindow(cd.ssbWindowClass, cd.ssbWindowName)
    ssbLeft, ssbTop, ssbRight, ssbBottom = win32gui.GetWindowRect(ssbWindow)

    # 3.2 如果窗口最小化了就把它恢复到正常
    win32gui.ShowWindow(ssbWindow,win32con.SW_SHOWNORMAL)

    # 3.3 把窗口激活
    win32gui.SetForegroundWindow(ssbWindow)

    # 3.4 把随申办窗口移动到（0，0）
    win32gui.MoveWindow(ssbWindow, cd.left, cd.top, abs(ssbRight - ssbLeft), abs(ssbBottom - ssbTop), cd.repaint)

    # 3.5 打开随申码
    logger.info("打开随申码...")
    time.sleep(cd.generalWaitTime * cd.generalTimeAdjust)   # 打开小程序还是有点耗时的，以后可以升级成用文字识别，能识别出“随申码”在跳出循环
    target_x = cd.left + cd.ssm_x
    target_y = cd.top + cd.ssm_y
    pyautogui.click(target_x, target_y)
    time.sleep(cd.generalWaitTime * cd.generalTimeAdjust)  # 打开图形（二维码）比较慢）

    # 截图处理
    getImage(cd.familyList[-1])

    # 3.6 打开亲属随申码主链接，进入亲属随申码列表，第一个是本人
    logger.info("打开亲属随申码清单...")
    target_x = cd.left + cd.ssm_main_link_x
    target_y = cd.top + cd.ssm_main_link_y 
    pyautogui.click(target_x, target_y)
    time.sleep(cd.generalWaitTime)

    # 3.7 循环处理亲属随申码
    for i in range(len(cd.familyList) - 1):
        time.sleep(cd.generalWaitTime)
        target_x = cd.left + cd.ssm_family_x
        target_y = cd.top + cd.ssm_family_start_y + cd.ssm_family_interval_y * i
        pyautogui.click(target_x, target_y)
        time.sleep(cd.generalWaitTime * cd.generalTimeAdjust) 
        getImage(cd.familyList[i])

        # 返回亲属随申码列表
        time.sleep(cd.generalWaitTime)
        target_x = cd.left + cd.ssm_main_link_x
        target_y = cd.top + cd.ssm_main_link_y + cd.ssm_family_adjust_y
        pyautogui.click(target_x, target_y)

    # 截图完成后，关闭随申办等窗口，微信窗口保留
    logger.info("关闭【随申码】和【随申办】窗口...")
    win32gui.PostMessage(ssbWindow, win32con.WM_CLOSE, 0, 0)
    win32gui.PostMessage(wxWindow, win32con.WM_CLOSE, 0, 0)
